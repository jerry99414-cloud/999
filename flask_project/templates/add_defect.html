@app.route("/add_defect/", methods=["GET", "POST"]) def add_defect(sheet_name): if request.method == "POST": defect = request.form.get("defect") reg = request.form.get("reg") conn = sqlite3.connect(DB_PATH) c = conn.cursor() c.execute( "INSERT INTO defects (sheet_name, defect, reg) VALUES (?, ?, ?)", (sheet_name, defect, reg) ) conn.commit() conn.close() return redirect(url_for("defects", sheet_name=sheet_name)) return render_template("add_defect.html", sheet_name=sheet_name)
